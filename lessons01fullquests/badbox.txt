function showGuestBook($root)
{

    $res = [];
    $fh = fopen($root, 'r');

    while (!feof($fh)) {
        $line = fgets($fh);
        $res[] .= $line;
    }
    return $res;
    fclose($fh);
}

function arrayDiffAlt(array $arrayA, array $arrayB): array
{
    $arrayC = [];
    $arrayD = [];
    $lenA = count($arrayA);
    $lenB = count($arrayB);
    for ($i = 0; $i < $lenA; $i++) {
        for ($j = 0; $j < $lenB; $j++) {
            if ($arrayA[$i] != $arrayB[$j]) {
                $arrayC[] = $arrayA[$i];
                break;
            }
        }
    }

    for ($i = 0; $i < $lenB; $i++) {
        for ($j = 0; $j < $lenA; $j++) {
            if ($arrayA[$i] != $arrayB[$j]) {
                $arrayD[] = $arrayB[$i];
                break;
            }
        }
    }
    $lenD = count($arrayD);
    for ($i = 0; $i < $lenD; $i++) {
        $arrayC[] = $arrayD[$i];
    }
    return $arrayC;
}
